Повний style.css

src/style.css — копіюй повністю 
/* src/style.css */
@import "tailwindcss";

@layer utilities {

  /* ── Центрування ── */
  .center     { @apply flex items-center justify-center; }
  .center-col { @apply flex flex-col items-center justify-center; }
  .center-x   { @apply flex justify-center; }
  .center-y   { @apply flex items-center; }
  .between    { @apply flex items-center justify-between; }

  /* ── Розміри ── */
  .full       { @apply w-full h-full; }
  .screen     { @apply w-screen h-screen; }
  .square-8   { @apply w-8 h-8; }
  .square-10  { @apply w-10 h-10; }
  .square-12  { @apply w-12 h-12; }
  .square-16  { @apply w-16 h-16; }

  /* ── Позиціонування ── */
  .overlay    { @apply absolute inset-0; }
  .fixed-full { @apply fixed inset-0; }
  .center-abs {
    @apply absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2;
  }

  /* ── Типографіка ── */
  .text-truncate {
    @apply truncate overflow-hidden whitespace-nowrap;
  }
  .gradient-text {
    @apply bg-clip-text text-transparent
           bg-gradient-to-r from-sky-400 to-violet-400;
  }

  /* ── Glassmorphism ── */
  .glass      { @apply backdrop-blur-md bg-white/10 border border-white/20; }
  .glass-dark { @apply backdrop-blur-md bg-black/30 border border-white/10; }

  /* ── Кнопки ── */
  .btn {
    @apply inline-flex items-center justify-center
           px-4 py-2 rounded-lg font-medium
           transition-all duration-200 cursor-pointer;
  }
  .btn-primary {
    @apply btn bg-sky-500 text-white
           hover:bg-sky-600 active:scale-95;
  }
  .btn-outline {
    @apply btn border border-sky-500 text-sky-500
           hover:bg-sky-500 hover:text-white active:scale-95;
  }
  .btn-ghost {
    @apply btn border border-white/10 text-gray-400
           hover:bg-white/5 active:scale-95;
  }

  /* ── Картки ── */
  .card {
    @apply rounded-xl border border-white/10
           bg-white/5 p-6 transition-all duration-300;
  }
  .card-hover {
    @apply card hover:border-white/20
           hover:shadow-xl hover:-translate-y-1;
  }

  /* ── Layout ── */
  .container-xl { @apply max-w-7xl mx-auto px-4 md:px-8; }
  .container-md { @apply max-w-3xl mx-auto px-4 md:px-6; }
  .section      { @apply py-16 md:py-24; }
  .section-sm   { @apply py-8 md:py-12; }
}

/* ── No scrollbar (нативний CSS) ── */
.no-scrollbar {
  scrollbar-width: none;
  -ms-overflow-style: none;
}
.no-scrollbar::-webkit-scrollbar {
  display: none;
}
